def portfolio_risk_and_reward(hist_data, weights):

    n = len(hist_data)
    x = weights
    T = len(hist_data[0])
    exp_return = [sum(r) / T for r in hist_data]
    reward = sum([x[j] * exp_return[j] for j in range(n)])

    sum_abs_dev = 0
    for t in range(T):
        dev = sum(x[j] * (hist_data[j][t] - exp_return[j]) for j in range(n))
        abs_dev = abs(dev)
        sum_abs_dev += abs_dev

    risk = sum_abs_dev / T

    return risk, reward
